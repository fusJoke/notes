     * @return bool

    public function pay(ChannelMerchantsBillPayForm $form) {
        $attributes['billId'] = $form->billId;
        $billOrder = $this->getBillOrderById($attributes);
        if(!$billOrder){
            return false;
        }

        if($this->checkExistPaymentOrder($form->billId)){
            return $this->setError('该付款单已提交，请勿重复提交');
        }

        $attributes['amount'] = $billOrder->payable_amount;
        $attributes['merchantsId'] = $this->getUser()->merchants_id;
        $attributes['transferMethod'] = $form->transferMethod;
        $attributes['transferUserName'] = $form->transferUserName;
        $attributes['transferDate'] = $form->transferDate;
        $attributes['receiptAccountNumber'] = $form->receiptAccountNumber;
        $attributes['receiptAccountBankName'] = $form->receiptAccountBankName;
        $attributes['note'] = $form->note;

        $res = ChannelMerchantsBillPaymentOrderModel::create($attributes);

        if (Error::isError($res)) {
            return $this->setError($res->getError());
        }

        return true;
    }



    public function getBillOrderById($id) {
        $order = ChannelMerchantsBill::findOne(['id' => $id]);

        return $order;
    }
