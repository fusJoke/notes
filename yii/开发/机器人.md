昵称 标识要搜索

操作启用、弃用



展示的字段： identity、head_img、nick_name、client_code

```sql
CREATE TABLE `robot` (
  `id` INT NOT NULL AUTO_INCREMENT COMMENT 'id',
  `identity` varchar(128) NOT NULL COMMENT '机器人标识',
  `head_img` varchar(128) NOT NULL COMMENT '机器⼈头像',
  `nick_name` varchar(128) NOT NULL COMMENT '机器⼈昵称',
  `client_code` varchar(128) NOT NULL COMMENT '机器⼈所属用户',
  `status` tinyint unsigned not null default 10 comment '机器人状态:10表示启用，20表示暂停',
  `wxid` varchar(128) NOT NULL COMMENT '机器⼈唯一标志',
  `create_time` DATETIME NOT NULL COMMENT '创建时间',
  `update_time` DATETIME NOT NULL COMMENT '更新时间',
  PRIMARY KEY (`id`)
) ENGINE=INNODB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_bin COMMENT='客服机器人';
```
